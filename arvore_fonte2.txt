<INCLUDE>  ::= "<" <LIB_NAME> "> <INCLUDE>
<LIB_NAME> ::= string.h

<S0> ::= ; <FUNCTION_>

<STATEMENT> ::= <DECL> <STATEMENT>
<DECL> ::= <TYPE> <VAR_LIST>
<TYPE> ::= INT
<VAR_LIST> :: <IDENTIFIER> <VAR_LIST_ROW>
<IDENTIFIER> ::= a
<VAR_LIST_ROW> :: ;
<STATEMENT> ::= 位
